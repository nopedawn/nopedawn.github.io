from sympy import nextprime, root
from Crypto.Util.number import long_to_bytes

def factorize(n):
    p = nextprime(int(root(n, 3)))
    while n % p != 0:
        p = nextprime(p)
    n //= p
    q = nextprime(p)
    while n % q != 0:
        q = nextprime(q)
    r = n // q
    return p, q, r

def decrypt(ct, p, q, r, e):
    phi = (p - 1) * (q - 1) * (r - 1)
    d = pow(e, -1, phi)
    return pow(ct, d, n)
