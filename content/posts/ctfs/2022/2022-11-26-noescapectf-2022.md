```python
def str_xor(secret, key):
    #extend key to secret length
    new_key = key
    i = 0
    while len(new_key) < len(secret):
      new_key = new_key + key[i]
      i = (i + 1) % len(key)
    a = []
    for (secret_c,new_key_c) in zip(secret,new_key):
        a.append(chr(ord(secret_c) ^ ord(new_key_c)))
